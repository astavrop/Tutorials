116 const TetrahedronWrapper<Point_t> * Delaunay3d<Point_t>::findTetrahedron(const Geometry::Vec3 & pos, float epsilon) {
117  validate();
118 
119  // Check the cache.
120  if(lastResult != nullptr && lastResult->getTetrahedron().containsPoint(pos, epsilon)) {
121  return lastResult;
122  }
123 
124  for(const auto & wrapper : tetrahedra) {
125  if(wrapper.getTetrahedron().containsPoint(pos, epsilon)) {
126  // Fill the cache.
127  lastResult = &wrapper;
128  return lastResult;
129  }
130  }
131  return nullptr;
132 }

134 template<typename Point_t>
135 const TetrahedronWrapper<Point_t> * Delaunay3d<Point_t>::findNearestTetrahedron(const Geometry::Vec3 & pos) {
136  validate();
137 
138  const TetrahedronWrapper<Point_t> * result = nullptr;
139  float minimumDistance = std::numeric_limits<float>::max();
140 
141  for(const auto & wrapper : tetrahedra) {
142  const float distance = wrapper.getTetrahedron().distanceSquared(pos);
143  if(distance < minimumDistance) {
144  minimumDistance = distance;
145  result = &wrapper;
146  }
147 
148  // Stop if the tetrahedron containing the position has been found.
149  if(minimumDistance == 0.0f) {
150  break;
151  }
152  }
153  return result;
154 }

156 template<typename Point_t>
157 void Delaunay3d<Point_t>::updateTetrahedra(){
158  // Clear the cache.
159  lastResult = nullptr;
160 
161  tetrahedra.clear();
162 
163  if(points.size() == 4) {
164  tetrahedra.emplace_back(&points[0], &points[1], &points[2], &points[3]);
165  } else if(points.size() > 4) {
166  std::vector<Geometry::Vec3f> positions;
167  positions.reserve(points.size());
168  for(const auto & point : points) {
169  positions.emplace_back(point.getPosition());
170  }
171  const auto results = doTetrahedralization(positions);
172  tetrahedra.reserve(results.size());
173  for(const auto & result : results) {
174  tetrahedra.emplace_back(&points[std::get<0>(result)], &points[std::get<1>(result)], &points[std::get<2>(result)], &points[std::get<3>(result)]);
175  }
176  }
177 
178  valid = true;
179 }
