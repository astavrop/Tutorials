169  bool IsInNeighbourhood(sPoint Point, float MinDist, float CellSize)
170  {
171  sGridPoint G = ImageToGrid(Point, CellSize);
172 
173  // number of adjucent cells to look for neighbour points
174  const int D = 5;
175 
176  // scan the neighbourhood of the point in the grid
177  for (int i = G.x - D; i < G.x + D; i++)
178  {
179  for (int j = G.y - D; j < G.y + D; j++)
180  {
181  if (i >= 0 && i < m_W && j >= 0 && j < m_H)
182  {
183  sPoint P = m_Grid[i][j];
184 
185  if (P.m_Valid && GetDistance(P, Point) < MinDist) { return true; }
186  }
187  }
188  }
189 
190 
191  return false;
192  }

211  template <typename PRNG>
212  sPoint GenerateRandomPointAround(const sPoint& P, float MinDist, PRNG& Generator)
213  {
214  // start with non-uniform distribution
215  float R1 = Generator.RandomFloat();
216  float R2 = Generator.RandomFloat();
217 
218  // radius should be between MinDist and 2 * MinDist
219  float Radius = MinDist * (R1 + 1.0f);
220 
221  // random angle
222  float Angle = 2 * 3.141592653589f * R2;
223 
224  // the new point is generated around the point (x, y)
225  float X = P.x + Radius * cos(Angle);
226  float Y = P.y + Radius * sin(Angle);
227 
228  return sPoint(X, Y);
229  }

238  template <typename PRNG = DefaultPRNG>
239  std::vector<sPoint> GeneratePoissonPoints(
240  size_t NumPoints,
241  PRNG& Generator,
242  int NewPointsCount = 30,
243  bool Circle = true,
244  float MinDist = -1.0f
245  )
246  {
247  if (MinDist < 0.0f)
248  {
249  MinDist = sqrt(float(NumPoints)) / float(NumPoints);
250  }
251 
252  std::vector<sPoint> SamplePoints;
253  std::vector<sPoint> ProcessList;
254 
255  // create the grid
256  float CellSize = MinDist / sqrt(2.0f);
257 
258  int GridW = static_cast<int>(ceil(1.0f / CellSize));
259  int GridH = static_cast<int>(ceil(1.0f / CellSize));
260 
261  sGrid Grid(GridW, GridH, CellSize);
262 
263  sPoint FirstPoint;
264  do {
265  FirstPoint = sPoint(Generator.RandomFloat(), Generator.RandomFloat());
266  } while (!(Circle ? FirstPoint.IsInCircle() : FirstPoint.IsInRectangle()));
267 
268  // update containers
269  ProcessList.push_back(FirstPoint);
270  SamplePoints.push_back(FirstPoint);
271  Grid.Insert(FirstPoint);
272 
273  // generate new points for each point in the queue
274  while (!ProcessList.empty() && SamplePoints.size() < NumPoints)
275  {
276 #if POISSON_PROGRESS_INDICATOR
277  // a progress indicator, kind of
278  if (SamplePoints.size() % 100 == 0) std::cout << ".";
279 #endif // POISSON_PROGRESS_INDICATOR
280 
281  sPoint Point = PopRandom<PRNG>(ProcessList, Generator);
282 
283  for (int i = 0; i < NewPointsCount; i++)
284  {
285  sPoint NewPoint = GenerateRandomPointAround(Point, MinDist, Generator);
286 
287  bool Fits = Circle ? NewPoint.IsInCircle() : NewPoint.IsInRectangle();
288 
289  if (Fits && !Grid.IsInNeighbourhood(NewPoint, MinDist, CellSize))
290  {
291  ProcessList.push_back(NewPoint);
292  SamplePoints.push_back(NewPoint);
293  Grid.Insert(NewPoint);
294  continue;
295  }
296  }
297  }
298 
299 #if POISSON_PROGRESS_INDICATOR
300  std::cout << std::endl << std::endl;
301 #endif // POISSON_PROGRESS_INDICATOR
302 
303  return SamplePoints;
304  }
