131  void split(uint32_t axis, float ratio) {
132  if(hasChildren()) {

136  Geometry::Vec3f mini(bounds.getMinX(), bounds.getMinY(), bounds.getMinZ());

138  Geometry::Vec3f mini2(maxi);

140  float f;
141  switch(axis) {
142  case 0:

144  mini2.setX(f);
145  maxi2.setX(f);
146  break;
147  case 1:
148  f = mini.getY()+bounds.getExtentY() * ratio;
149  mini2.setY(f);
150  maxi2.setY(f);
151  break;
152  case 2:
153  f = mini.getZ()+bounds.getExtentZ() * ratio;
154  mini2.setZ(f);
155  maxi2.setZ(f);
156  break;
157  default:
158  FAIL();
159  }
160  children.emplace_back(new Region(Geometry::Box(mini,mini2), this));
161  children.emplace_back(new Region(Geometry::Box(maxi,maxi2), this));
162  children.shrink_to_fit();
