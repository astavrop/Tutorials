74  float countPolygons(const Geometry::Box & box) {
75  const auto geoNodes = collectGeoNodesIntersectingBox(sceneRoot.get(), box);
76  float approx = 0;
77  for(const auto & geoNode : geoNodes) {
78  const auto b = Geometry::Intersection::getBoxBoxIntersection(geoNode->getWorldBB(), box);
79  if(b.isValid() && b.getExtentMin() > 0)
80  approx += geoNode->getTriangleCount() / geoNode->getWorldBB().getVolume() * b.getVolume();
81  }
82  return approx;
83  }
