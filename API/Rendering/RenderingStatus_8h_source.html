108  const LightParameters & getEnabledLight(uint8_t index) const {
109  assert(index < getNumEnabledLights());
110  uint_fast8_t pos = 0;
111  // Find first enabled light.
112  while(!lightsEnabled[pos]) {
113  ++pos;
114  }
115  // Find next enabled light, until the index'th enabled light is found.
116  while(index > 0) {
117  while(!lightsEnabled[pos]) {
118  ++pos;
119  }
120  --index;
121  ++pos;
122  }
123  return lights[pos];
124  }
126  uint8_t enableLight(const LightParameters & light) {
127  assert(getNumEnabledLights() < MAX_LIGHTS);
128  uint_fast8_t pos = 0;
129  while(lightsEnabled[pos]) {
130  ++pos;
131  }
132  ++lightsCheckNumber;
133  lights[pos] = light;
134  lightsEnabled[pos] = true;
135  return pos;
136  }
138  void disableLight(uint8_t lightNumber) {
139  assert(lightsEnabled[lightNumber]);
140  ++lightsCheckNumber;
141  lightsEnabled[lightNumber] = false;
142  }
144  bool isLightEnabled(uint8_t lightNumber) const {
145  return lightsEnabled[lightNumber];
146  }
147  bool lightsChanged(const RenderingStatus & actual) const {
148  if (lightsCheckNumber == actual.lightsCheckNumber)
149  return false;
150  if (lightsEnabled != actual.lightsEnabled)
151  return true;
