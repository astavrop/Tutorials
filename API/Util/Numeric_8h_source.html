55 inline bool invertMatrix(float * a, uint16_t n) {
56  const float epsilon = 0.0001f; // Genauigkeit
57  const uint_fast16_t rowSize = 2u * n;
58 
59  // Set right half of matrix to identity.
60  for (uint_fast16_t i = 0; i < n; ++i) {
61  for (uint_fast16_t j = 0; j < n; ++j) {
62  const uint_fast16_t elementOffset = i * rowSize + n + j;
63  if (i == j) {
64  a[elementOffset] = 1.0f;
65  } else {
66  a[elementOffset] = 0.0f;
67  }
68  }
69  }
70 
71  // gauss-jordan-algorithm
72  for(uint_fast16_t s = 0; s < n; ++s) {
73  const uint32_t rowOffsetS = s * rowSize;
74 
75  // Pivotisierung vermeidet unnï¿½tigen Abbruch bei einer Null in der Diagnonalen und
76  // erhï¿½ht die Rechengenauigkeit
77  float maximum = std::abs(a[rowOffsetS + s]);
78 
79  uint_fast16_t prow = s;
80  for (uint_fast16_t i = s+1; i < n; ++i) {
81  const uint32_t rowOffsetI = i * rowSize;
82  if (std::abs(a[rowOffsetI + s]) > maximum) {
83  maximum = std::abs(a[rowOffsetI + s]);
84  prow = i;
85  }
86  }
87 
88  if(maximum < epsilon) {
89  return false; // not invertible
90  }
91 
92  if (prow != s){ // swap rows if necessary
93  for (uint_fast16_t j = s; j < rowSize; ++j){
94  float h = a[s * rowSize + j];
95  a[s * rowSize + j] = a[prow * rowSize + j];
96  a[prow * rowSize + j]= h;
97  }
98  }
99 
100  // divide row by pivot coefficient f = a[s][s]
101  float f = a[rowOffsetS + s];
102  for (uint_fast16_t j = s; j < rowSize; ++j) {
103  a[rowOffsetS + j] /= f;
104  }
105 
106  // elimination: make column s a unit vector by adding row s to row i
107  for (uint_fast16_t i = 0; i < n; ++i){
108  if (i != s){
109  const uint32_t rowOffsetI = i * rowSize;
110  f = -a[rowOffsetI + s];
111  for (uint_fast16_t j = s; j < rowSize; ++j) {
112  a[rowOffsetI + j] += f*a[rowOffsetS + j];
113  }
114  }
115  }
116  }
