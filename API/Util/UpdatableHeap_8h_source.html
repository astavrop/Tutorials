147 inline void UpdatableHeap<Key_t, Data_t>::erase(UpdatableHeapElement* h){
148  --elementCount;
149 
150  UpdatableHeapElement *tmp, *last;
151  tmp = heapTail.back();
152  heapTail.pop_back();
153  delete tmp;
154  tmp = heapTail.back();
155  heapTail.pop_back();
156  heapTail.push_front(tmp);
157 
158  // swap last and h
159  const Key_t oldKey = h->cost;
160  last = tmp->parent;
161  if(last!=h){
162  // use dummy to replace "last"
163  if(last->parent->h1 == last){
164  last->parent->h1 = tmp;
165  }else{
166  last->parent->h2 = tmp;
167  }
168  tmp->parent = last->parent;
169 
170  last->h1 = h->h1;
171  last->h1->parent = last;
172 
173  last->h2 = h->h2;
174  last->h2->parent = last;
175 
176  last->parent = h->parent;
177  if(last->parent!=nullptr){
178  if(last->parent->h1 == h){
179  last->parent->h1 = last;
180  }else{
181  last->parent->h2 = last;
182  }
183  }
184  }else{
185  if(h->parent!=nullptr){
186  if(h->parent->h1 == h){
187  h->parent->h1 = tmp;
188  }else{
189  h->parent->h2 = tmp;
190  }
191  }
192  tmp->parent = h->parent;
193  last = tmp;
194  }
195 
196  if(root==h)
197  root = last;
198 
199  delete h;
200 
201  if(last->getCost() < oldKey) {
202  moveUpHeap(last);
203  } else {
204  moveDownHeap(last);
205  }
206 }

209 inline void UpdatableHeap<Key_t, Data_t>::moveDownHeap(UpdatableHeapElement* h){
210  if(h->dummy)
211  return;
212  if(h==root){
213  if(!h->h1->dummy){
214  if(!h->h2->dummy){
215  // this has two children
216  if(h->cost > h->h1->cost && h->h2->cost >= h->h1->cost){
217  // swap this and h1
218  root = h->h1;
219  swapHeapElements(h, h->h1);
220  }else{
221  if(h->cost > h->h2->cost){
222  // swap this and h2
223  root = h->h2;
224  swapHeapElements(h, h->h2);
225  }
226  }
227  }else{
228  // this has one child
229  if(h->cost > h->h1->cost){
230  // swap this and h1
231  root = h->h1;
232  swapHeapElements(h, h->h1);
233  }
234  }
235  }
236  }
237  bool moved;
238  do{
239  moved = false;
240  if(!h->h1->dummy){
241  if(!h->h2->dummy){
242  // this has two children
243  if(h->cost > h->h1->cost && h->h2->cost >= h->h1->cost){
244  // swap this and h1
245  moved = true;
246  swapHeapElements(h, h->h1);
247  }else{
248  if(h->cost > h->h2->cost){
249  // swap this and h2
250  moved = true;
251  swapHeapElements(h, h->h2);
252  }
253  }
254  }else{
255  // this has one child
256  if(h->cost > h->h1->cost){
257  // swap this and h1
258  moved = true;
259  swapHeapElements(h, h->h1);
260  }
261  }
262  }

268  if(h->dummy)
269  return;
270  bool moved;
271  do{
272  moved = false;
273  if(h->parent!=nullptr && h->parent->cost > h->cost){
274  // parent has higher cost -> swap
275  moved = true;
276  swapHeapElements(h->parent, h);
277  }
278  }while(moved);
279 
280  if(h->parent == nullptr)
281  root = h;
282 }

285 inline void UpdatableHeap<Key_t, Data_t>::swapHeapElements(UpdatableHeapElement *hp, UpdatableHeapElement *hc){
286  if(hp->h1 == hc) {
287  /* (hp)
288  * h1 / \ h2
289  * / \
290  * (hc) ( )
291  */
292  // swap child h1
293  hc->h1->parent = hp;
294  hp->h1 = hc->h1;
295  hc->h1 = hp;
296 
297  // swap child h2
298  hp->h2->parent = hc;
299  hc->h2->parent = hp;
300  using std::swap;
301  swap(hp->h2, hc->h2);
302  } else {
303  /* (hp)
304  * h1 / \ h2
305  * / \
306  * ( ) (hc)
307  */
308  // swap child h2
309  hc->h2->parent = hp;
310  hp->h2 = hc->h2;
311  hc->h2 = hp;
312 
313  // swap child h1
314  hp->h1->parent = hc;
315  hc->h1->parent = hp;
316  using std::swap;
317  swap(hp->h1, hc->h1);
318  }
319 
320  // set hc as new child of hp's parent
321  if(hp->parent != nullptr) {
322  if(hp->parent->h1 == hp){
323  hp->parent->h1 = hc;
324  } else {
325  hp->parent->h2 = hc;
326  }
327  }
328 
329  // set hp's and hc's parents
330  hc->parent = hp->parent;
331  hp->parent = hc;
332 }
