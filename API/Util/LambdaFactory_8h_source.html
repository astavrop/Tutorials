60 template < class ObjectType,
61  typename IdentifierType,
62  template<class, typename> class FallbackPolicy = FallbackPolicies::ExceptionFallback >
63 class LambdaFactory {
64  private:
65  struct Creator final {
66  void* function;
67  const std::type_info* signature;
68  };
69  typedef std::map<IdentifierType, Creator> registrations_t;
70  registrations_t registrations;
71  public:
72  typedef FallbackPolicy<ObjectType, IdentifierType> fallbackPolicy_t;
73  fallbackPolicy_t fallbackPolicy;
74  LambdaFactory() : fallbackPolicy() {
75  }
76  LambdaFactory(fallbackPolicy_t policy) : fallbackPolicy(policy) {
77  }
78  ~LambdaFactory() {
79  for(auto entry : registrations)
80  delete static_cast<std::function<ObjectType()>*>(entry.second.function);
81  }

83  template<typename ObjectCreator>
84  void registerType(const IdentifierType & id, ObjectCreator creator) {
85  if(registrations.find(id) != registrations.end())
86  throw std::invalid_argument("the creator already exists");
87 
88  auto function = new decltype(to_function(creator))(to_function(creator));
89 
90  registrations[id].function = static_cast<void*>(function);
91  registrations[id].signature = &typeid(function);
92  }
93 
94  void unregisterType(const IdentifierType & id) {
95  if(registrations.find(id) == registrations.end())
96  return;
97  delete static_cast<std::function<ObjectType()>*>(registrations[id].function);
98  registrations.erase(id);
99  }
100 
101  template<typename ...Args>
102  ObjectType create(const IdentifierType & id, Args... args) {
103  if(registrations.find(id) == registrations.end())
104  return fallbackPolicy.onUnknownType(std::bind(&LambdaFactory::create<>, this, std::placeholders::_1), id);
105 
106  auto creator = registrations.at(id);
107  auto function = static_cast<std::function<ObjectType(Args...)>*>(creator.function);
108 
109  if(typeid(function) != *(creator.signature))
110  throw std::bad_typeid();
111 
112  return (*function)(args...);
113  }
114 };
