149  uint64_t getNanosecondsSinceReset() const {
150  timer_t time;
151  timer_t result;
152  queryTime(&time);
153 #ifdef _WIN32
154  result.QuadPart = time.QuadPart - lastReset.QuadPart;
155  return static_cast<uint64_t>(static_cast<double>(result.QuadPart) / static_cast<double>(frequency.QuadPart) * 1e9);
156 #elif (defined(__USE_POSIX199309) && defined(_POSIX_MONOTONIC_CLOCK)) || defined(ANDROID)
157  result.tv_sec = time.tv_sec - lastReset.tv_sec;
158  result.tv_nsec = time.tv_nsec - lastReset.tv_nsec;
159  if (result.tv_nsec < 0) {
160  --result.tv_sec;
161  result.tv_nsec += 1000000000L;
162  }
163  return static_cast<uint64_t>(result.tv_nsec) + 1000000000L * static_cast<uint64_t>(result.tv_sec);
164 #else
165  timersub(&time, &lastReset, &result);
166  return 1000L * static_cast<uint64_t>(result.tv_usec) + 1000000000L * static_cast<uint64_t>(result.tv_sec);
167 #endif
168  }
