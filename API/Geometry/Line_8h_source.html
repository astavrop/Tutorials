153  std::pair<value_t, value_t> getClosestValues(const line_t & lineB) const {
154  const value_t zero = static_cast<value_t>(0);
155  const value_t one = static_cast<value_t>(1);
156  const line_t & lineA = *this;

158  /*
159  * We have to minimize a function of the form f(x, y) = a xÂ² + 2 b x y + c yÂ² + 2 d x + 2 e y + f.
160  * The gradient is âf(x, y) = (Î´f/Î´x, Î´f/Î´y) = (2 a x + 2 b y + 2 d, 2 c y + 2 b x + 2 e).
161  * The Hessian matrix is H(f) = ( (Î´Â²f/(Î´x Î´x), Î´Â²f/(Î´x Î´y)), (Î´Â²f/(Î´y Î´x), Î´Â²f/(Î´y Î´y)) ) = ( (2 a, 2 b), (2 b,
162  * 2 c) ).
163  * The determinant of H(f) is 2 a * 2 c - 2 b * 2 b = 4 (a c - bÂ²).
164  * The roots of the gradient are x = - (b e - c d) / (a c - bÂ²) and y = - (b d - a e) / (a c - bÂ²).
165  */
166  // a = lineA.getDirection() * lineA.getDirection() = 1 (because directions of lines have unit length)
167  const value_t b = -lineA.getDirection().dot(lineB.getDirection());
168  // c = lineB.getDirection() * lineB.getDirection() = 1 (because directions of lines have unit length)
169  const value_t det = one - b * b; // ignore the 4 here, because we are interested in the sign only
170 
171  const vec_t diff = lineA.getOrigin() - lineB.getOrigin();
172  const value_t d = lineA.getDirection().dot(diff);
173 
174  if (det >= std::numeric_limits<value_t>::epsilon()) {
175  // The lines are not parallel.
176  const value_t e = -lineB.getDirection().dot(diff);
177  const value_t invDet = one / det;
178  return std::make_pair((b * e - d) * invDet, // parameterA: drop c here because it is 1
179  (b * d - e) * invDet); // parameterB: drop a here because it is 1
180  } else {
181  // The lines are parallel. Pick a pair arbitrary matching closest pair of points.
182  return std::make_pair(-d, zero);
183  }
