88  return (pos.x() >= origin.x() + hSideLength ? 1 : 0) + (pos.y() >= origin.y() + hSideLength ? 2 : 0)

91  Box_t getOctant(uint8_t i) const {
92  const auto hSideLength = static_cast<integer_t>(sideLength) / 2;
93  const Vec3_t minPos(origin.x() + ((i & 1) > 0 ? hSideLength : 0),
94  origin.y() + ((i & 2) > 0 ? hSideLength : 0),
95  origin.z() + ((i & 4) > 0 ? hSideLength : 0));

157  Area * insertIntermediateChild(const Vec3_t _origin, uinteger_t _sideLength) {
158  assert(_sideLength < sideLength);
159  assert(contains(_origin));
160  auto & children = assureContainer();
161  const uint8_t childIndex = getChildIndex(_origin);
162  Area * newChild = new Area(_origin, _sideLength, uniformValue);
163  Area * oldChild = children[childIndex];
164  children[childIndex] = newChild;

213  block_t & findOrCreateBlock(const Vec3_t & pos) {
214  // std::cout << "findOrCreateBlock("<<pos<<")"<<std::endl;
215  if (!root) {
216  root.reset(new Area(calcOrigin(pos, blockSideLength), blockSideLength, nullVoxel));
217  // std::cout << "creating root: "<<root->origin<<std::endl;
218  }

221  const auto commonBox = getEnclosingAreaBox(*root.get(), pos);
222  // std::cout << "resetting root: "<<commonBox.first<<" : "<<commonBox.second<<std::endl;
223  Area * newRoot = new Area(commonBox.first, commonBox.second, nullVoxel);
224  Area * oldRoot = root.release();
225  newRoot->setChild(newRoot->getChildIndex(oldRoot->getOrigin()), oldRoot);
226  root.reset(newRoot);
227  }

229  Area * currentArea = root.get();
230 
231  while (true) {
232  // std::cout << " >("<<currentArea->origin<<" : "<<currentArea->sideLength<< ")"<<std::endl;
233  currentArea->markedForConsolidation = true;

237  Area * child = currentArea->getChild(currentArea->getChildIndex(pos));

239  currentArea =

241  } else if (child->contains(pos)) {
242  currentArea = child;
243  } else { // insert intermediate area

251  //"<<std::endl;
252 
253  currentArea = currentArea->insertIntermediateChild(commonBox.first, commonBox.second);
254  }
255  } else { // if(currentArea->isUniform()){
256  if (currentArea->sideLength == blockSideLength) {
257  return currentArea->assureBlock();
258  } else {
259  currentArea =

265  void consolidate(Area * area) {
266  if (area && area->markedForConsolidation) {
267  area->markedForConsolidation = false;
268 
269  if (area->isBlock()) {
270  const auto & block = area->assureBlock();
271  Voxel_t value = block[0];
272  for (uint32_t j = 1; j < blockSize; ++j) {
273  if (block[j] != value)
274  return; // simplification not possible
275  }
276  area->convertToUniformArea(value);
277  } else if (area->isContainer()) {
278  Voxel_t value = area->uniformValue;
279  bool childrenAreUniform = true;
280 
281  for (uint8_t i = 0; i < 8; ++i) {
282  Area * child = area->getChild(i);
283  if (child && child->markedForConsolidation)
284  consolidate(child);
285  if (i == 0) { // first child
286  if (child) {
287  if (child->isUniform()) {
288  value = child->uniformValue;
289  } else {
290  childrenAreUniform = false;
291  }
292  } // else keep area's uniform value
293  } else if (child) {
294  if (childrenAreUniform && (!child->isUniform() || child->uniformValue != value))
295  childrenAreUniform = false;
296  } else { // else !child
297  if (childrenAreUniform && area->uniformValue != value)
298  childrenAreUniform = false;
299  }
300  }
301  if (childrenAreUniform) {
302  area->convertToUniformArea(value);
303  // std::cout << "Consolidated area: "<<area->getBox()<<"\n";
304  }
305  } // else area->isUniform()
306  }

348  void fill(const Box_t & fillArea, const Voxel_t & voxel) {
349  // std::cout << "fill:"<<fillArea<<" "<<std::endl;
350 
351  // assure properly sized root node.
352  if (!root || !root->getBox().contains(fillArea)) {
353  findOrCreateBlock(fillArea.getMin());
354  findOrCreateBlock(fillArea.getMax());
355  }
356 
357  // std::cout << "start filling "<<std::endl;
358 
359  std::stack<Area *> todo;
360  todo.push(root.get());
361  while (!todo.empty()) {
362  Area * currentArea = todo.top();
363  todo.pop();
364  // std::cout << " > "<<currentArea->getBox()<<std::endl;
365 
366  const Box_t areaBox(currentArea->getBox());
367  if (fillArea.contains(areaBox)) {
368  currentArea->convertToUniformArea(voxel);
369  // std::cout << " >F "<< currentArea->origin<<" : "<<currentArea->sideLength
370  //<<std::endl;
371  // std::cout << " >F "<< currentArea->sideLength;
372  } else if (Intersection::isBoxIntersectingBox(areaBox, fillArea)) {
373  currentArea->markedForConsolidation = true;
374  if (currentArea->isBlock() || currentArea->sideLength == blockSideLength) {
375  const Box_t intersection = Intersection::getBoxBoxIntersection(areaBox, fillArea);
376  block_t & block = currentArea->assureBlock();
377  for (integer_t x = intersection.getMinX(); x <= intersection.getMaxX(); ++x) {
378  for (integer_t y = intersection.getMinY(); y <= intersection.getMaxY(); ++y) {
379  for (integer_t z = intersection.getMinZ(); z <= intersection.getMaxZ(); ++z) {
380  block[posToBlockIdx(Vec3_t(x, y, z))] = voxel;
381  }
382  }
383  }
384  // std::cout << " >B "<< currentArea->origin<<" : "<<currentArea->sideLength
385  //<<std::endl;
386  } else { // uniform value or container
387  currentArea->assureContainer();
388  for (uint8_t i = 0; i < 8; ++i) {
389  const Box_t octantIntersection =
390  Intersection::getBoxBoxIntersection(currentArea->getOctant(i), fillArea);
391  if (octantIntersection.isInvalid()) // no intersection?
392  continue;
393  Area * child = currentArea->getChild(i);
394  if (child && child->getBox().contains(octantIntersection)) { // fill inside existing child...
395  todo.push(child);
396  } else if (!child) { // create new child covering the filling
397  const auto enclosingBox = getEnclosingAreaBox(octantIntersection);
398  // std::cout << " >>a: \t"<< currentArea->getOrigin()
399  //<<std::endl;
400  // std::cout << " >>octant:"<<static_cast<int>(i)<<" \t"<<
401  //currentArea->getOctant(i) <<std::endl;
402  // std::cout << " >>octantIntersection:\t"<< octantIntersection
403  //<<std::endl;
404  // std::cout << " >>enclosingBox:\t"<< enclosingBox.first<<" :
405  //"<<enclosingBox.second <<std::endl;
406  todo.push(currentArea->insertIntermediateChild(enclosingBox.first, enclosingBox.second));
407  } else { // create an intermediate child covering the filled area and the old child
408  // std::cout << " B ";
409  // std::cout << "\n >>
410  //octantIntersection:\t"<<octantIntersection;
411  Box_t b = octantIntersection;
412  b.include(child->getBox());
413  // std::cout << "\n >> childBox:\t"<<child->getBox();
414  const auto enclosingBox = getEnclosingAreaBox(b);
415  // std::cout << "\n >> combined:\t"<<b;
416  // std::cout << "\n >> area:\t"<<enclosingBox.first<<" :
417  //"<<enclosingBox.second<<"\n";
418  todo.push(currentArea->insertIntermediateChild(enclosingBox.first, enclosingBox.second));
419  }
420  }
421  }
422  continue;
423  } // else not intersecting -> skip
424  }
425  consolidate(root.get());
426  }

440  serializationData_t serialize(const Box_t & queryBox) {
441  std::vector<std::tuple<Vec3_t, uinteger_t, Voxel_t>> uniformAreas;
442  std::vector<std::tuple<Vec3_t, block_t>> singleValues;
443 
444  consolidate(root.get());
445  std::stack<Area *> todo;
446  if (root)
447  todo.push(root.get());
448  while (!todo.empty()) {
449  Area * currentArea = todo.top();
450  todo.pop();
451 
452  if (!Intersection::isBoxIntersectingBox(queryBox, currentArea->getBox()))
453  continue;
454  if (currentArea->isBlock()) {
455  singleValues.emplace_back(currentArea->getOrigin(), currentArea->assureBlock());
456  } else if (currentArea->isContainer()) {
457  for (uint8_t i = 0; i < 8; ++i) {
458  Area * child = currentArea->getChild(i);
459  if (child)
460  todo.push(child);
461  }
462  } else { // currentArea->isUniform
463  if (currentArea->uniformValue != nullVoxel) // skip null
464  uniformAreas.emplace_back(currentArea->getOrigin(), currentArea->sideLength,
465  currentArea->uniformValue);
466  }
467  }
468  return std::make_pair(uniformAreas, singleValues);
469  }

471  void deserialize(const serializationData_t & data) {
472  for (const auto & uniformArea : data.first) {
473  const Box_t b(std::get<0>(uniformArea),
474  std::get<0>(uniformArea) + Vec3_t(std::get<1>(uniformArea), std::get<1>(uniformArea),
475  std::get<1>(uniformArea)));
476  fill(b, std::get<2>(uniformArea));
477  }
478  for (const auto & blockData : data.second)
479  findOrCreateBlock(std::get<0>(blockData)) = std::get<1>(blockData);
480  consolidate(root.get());

483  Box_t getBlockBounds() const {
484  Box_t b;
485  b.invalidate();
486  std::stack<const Area *> todo;
487  if (root)
488  todo.push(root.get());
489  while (!todo.empty()) {
490  const Area * currentArea = todo.top();
491  todo.pop();
492  if (currentArea->isContainer()) {
493  for (uint8_t i = 0; i < 8; ++i) {
494  Area * child = currentArea->getChild(i);
495  if (child && !b.contains(child->getBox()))
496  todo.push(child);
497  }
498  } else if (currentArea->isBlock() || currentArea->uniformValue != nullVoxel) {
499  b.include(currentArea->getBox());
