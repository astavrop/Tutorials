125  static uint16_t floatToHalf(float value) {
126  static int32_t const infN = 0x7F800000; // flt32 infinity
127  static int32_t const maxN = 0x477FE000; // max flt16 normal as a flt32
128  static int32_t const minN = 0x38800000; // min flt16 normal as a flt32
129  static int32_t const nanN = 0x7f802000; // minimum flt16 nan as a flt32
130  union {
131  float f;
132  int32_t si;
133  uint32_t ui;
134  } v, s;
135  v.f = value;
136  uint32_t sign = v.si & 0x80000000; // flt32 sign bit
137  v.si ^= sign;
138  sign >>= 16; // logical shift
139  s.si = 0x52000000; // (1 << 23) / minN
140  s.si = s.f * v.f; // correct subnormals
141  v.si ^= (s.si ^ v.si) & -(minN > v.si);
142  v.si ^= (infN ^ v.si) & -((infN > v.si) & (v.si > maxN));
143  v.si ^= (nanN ^ v.si) & -((nanN > v.si) & (v.si > infN));
144  v.ui >>= 13; // logical shift
145  v.si ^= ((v.si - 0x1c000) ^ v.si) & -(v.si > 0x23bff);
146  v.si ^= ((v.si - 0x1c000) ^ v.si) & -(v.si > 0x003ff);
147  return v.ui | sign;
148  }

162  } v, s;

168  v.si ^= ((v.si + 0x1c000) ^ v.si) & -(v.si > 0x23bff);
169  s.si = 0x33800000; // minN / (1 << (23 - shift))
170  s.f *= v.si;
