110  void getRotation(vec3_t & axis, angle_t & angle) const {
111  axis.setX(at(R2C1) - at(R1C2));
112  axis.setY(at(R0C2) - at(R2C0));
113  axis.setZ(at(R1C0) - at(R0C1));
114  value_t length = axis.length();
115  value_t diagonal = at(R0C0) + at(R1C1) + at(R2C2);
116  const float rad = std::atan2(length, diagonal - 1.0f);
117  if (rad == 0.0f) {
118  axis.setValue(0.0f, 0.0f, 0.0f);
119  } else {
120  axis.normalize();
121  }
122  angle = angle_t::rad(rad);
123  }

183  const vec3_t f = r.cross(u).normalize();

191  static _Matrix3x3 createRotation(const angle_t & angle, const vec3_t & axis) {
192  const value_t c = std::cos(angle.rad());
193  const value_t s = std::sin(angle.rad());
194  const value_t t = 1 - c;
195  const value_t xx = axis.getX() * axis.getX();
196  const value_t xy = axis.getX() * axis.getY();
197  const value_t xz = axis.getX() * axis.getZ();
198  const value_t yy = axis.getY() * axis.getY();
199  const value_t yz = axis.getY() * axis.getZ();
200  const value_t zz = axis.getZ() * axis.getZ();
201  return _Matrix3x3(t * xx + c, t * xy - s * axis.getZ(), t * xz + s * axis.getY(), t * xy + s * axis.getZ(),
202  t * yy + c, t * yz - s * axis.getX(), t * xz - s * axis.getY(), t * yz + s * axis.getX(),
203  t * zz + c);
204  }

223  _Matrix3x3 getInverse() const {
224  const value_t d = det();

226  std::cerr << "Matrix not invertible, det = " << d << std::endl;
227  return *this;

231  (at(R1C1) * at(R2C2) - at(R1C2) * at(R2C1)) * dInv, -(at(R0C1) * at(R2C2) - at(R2C1) * at(R0C2)) * dInv,
