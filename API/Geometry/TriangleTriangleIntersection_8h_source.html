18 /* this edge to edge test is based on Franlin Antonio's gem:
19  "Faster Line Segment Intersection", in Graphics Gems III,
20  pp. 199-202 */
21 #define EDGE_EDGE_TEST(V0, U0, U1) \
22  B = U0 - U1; \
23  C = V0 - U0; \
24  f = A.y() * B.x() - A.x() * B.y(); \
25  d = B.y() * C.x() - B.x() * C.y(); \
26  if ((f > 0 && d >= 0 && d <= f) || (f < 0 && d <= 0 && d >= f)) { \
27  e = A.x() * C.y() - A.y() * C.x(); \
28  if (f > 0) { \
29  if (e >= 0 && e <= f) \
30  return 0; \
31  } else { \
32  if (e <= 0 && e >= f) \
33  return 0; \
34  } \
35  }

37 #define EDGE_AGAINST_TRI_EDGES(V0, V1, U0, U1, U2) \
38  { \
39  _Vec2<value_t> A, B, C; \
40  float e, d, f; \
41  A = V1 - V0; \
42  /* test edge U0,U1 against V0,V1 */ \
43  EDGE_EDGE_TEST(V0, U0, U1); \
44  /* test edge U1,U2 against V0,V1 */ \
45  EDGE_EDGE_TEST(V0, U1, U2); \
46  /* test edge U2,U1 against V0,V1 */ \
47  EDGE_EDGE_TEST(V0, U2, U0); \
48  }

50 #define POINT_IN_TRI(V0, U0, U1, U2) \
51  { \
52  float a, b, c, d0, d1, d2; \
53  /* is T1 completly inside T2? */ \
54  /* check if V0 is inside tri(U0,U1,U2) */ \
55  a = U1[1] - U0[1]; \
56  b = -(U1[0] - U0[0]); \
57  c = -a * U0[0] - b * U0[1]; \
58  d0 = a * V0[0] + b * V0[1] + c; \
59  \
60  a = U2[1] - U1[1]; \
61  b = -(U2[0] - U1[0]); \
62  c = -a * U1[0] - b * U1[1]; \
63  d1 = a * V0[0] + b * V0[1] + c; \
64  \
65  a = U0[1] - U2[1]; \
66  b = -(U0[0] - U2[0]); \
67  c = -a * U2[0] - b * U2[1]; \
68  d2 = a * V0[0] + b * V0[1] + c; \
69  if (d0 * d1 > 0.0) { \
70  if (d0 * d2 > 0.0) \
71  return 0; \
72  } \
73  }

78 template <typename value_t>
79 inline void _intersect(const _Vec3<value_t> & v0, const _Vec3<value_t> & v1, const _Vec3<value_t> & v2, value_t pv0,
80  value_t pv1, value_t pv2, float dv0, float dv1, float dv2, float * intersection1,
81  float * intersection2, _Vec3<value_t> & point1, _Vec3<value_t> & point2) {
82  float tmp = dv0 / (dv0 - dv1);
83  *intersection1 = pv0 + (pv1 - pv0) * tmp;
84  point1 += (v0 + (v1 - v0) * tmp);
85  tmp = dv0 / (dv0 - dv2);
86  *intersection2 = pv0 + (pv2 - pv0) * tmp;
87  point2 += (v0 + (v2 - v0) * tmp);
88 }

90 template <typename value_t>
91 inline bool _computeIntersectionInterval(const Triangle<_Vec3<value_t>> & triangle, float pv0, float pv1, float pv2,
92  float dv0, float dv1, float dv2, float dv0dv1, float dv0dv2,
93  float * intersection1, float * intersection2, _Vec3<value_t> & point1,
94  _Vec3<value_t> & point2) {
95  auto v0 = triangle.getVertexA();
96  auto v1 = triangle.getVertexB();
97  auto v2 = triangle.getVertexC();
98  if (dv0dv1 > 0.0f) {
99  // here we know that dv0dv2<=0.0
100  // that is dv0, dv1 are on the same side, dv2 on the other or on the plane
101  _intersect(v2, v0, v1, pv2, pv0, pv1, dv2, dv0, dv1, intersection1, intersection2, point1, point2);
102  } else if (dv0dv2 > 0.0f) {
103  // here we know that dv0dv1<=0.0
104  // that is dv0, dv2 are on the same side, dv1 on the other or on the plane
105  _intersect(v1, v0, v2, pv1, pv0, pv2, dv1, dv0, dv2, intersection1, intersection2, point1, point2);
106  } else if (dv1 * dv2 > 0.0f || dv0 != 0.0f) {
107  // here we know that dv0dv1<=0.0 or dv0!=0.0
108  // that is dv1, dv2 are on the same side, dv0 on the other or on the plane
109  _intersect(v0, v1, v2, pv0, pv1, pv2, dv0, dv1, dv2, intersection1, intersection2, point1, point2);
110  } else if (dv1 != 0.0f) {
111  _intersect(v1, v0, v2, pv1, pv0, pv2, dv1, dv0, dv2, intersection1, intersection2, point1, point2);
112  } else if (dv2 != 0.0f) {
113  _intersect(v2, v0, v1, pv2, pv0, pv1, dv2, dv0, dv1, intersection1, intersection2, point1, point2);
114  } else {
115  // triangles are coplanar
116  return true;
117  }
118  return false;
119 }

136 template <typename value_t>
137 int32_t getTriangleTriangleIntersection(const Triangle<_Vec3<value_t>> & triangle1,
138  const Triangle<_Vec3<value_t>> & triangle2,
139  _Segment<_Vec3<value_t>> & tSegment) {
140  const auto epsilon = std::numeric_limits<value_t>::epsilon();
141 
142  // compute plane of triangle2 and compute signed distances of triangle1 to the plane
143  const _Plane<value_t> plane2(triangle2.getVertexA(), triangle2.getVertexB(), triangle2.getVertexC());
144  float du0 = plane2.planeTest(triangle1.getVertexA());
145  float du1 = plane2.planeTest(triangle1.getVertexB());
146  float du2 = plane2.planeTest(triangle1.getVertexC());
147 
148  // coplanarity robustness check
149  if (std::fabs(du0) < epsilon)
150  du0 = 0.0;
151  if (std::fabs(du1) < epsilon)
152  du1 = 0.0;
153  if (std::fabs(du2) < epsilon)
154  du2 = 0.0;
155 
156  float du0du1 = du0 * du1;
157  float du0du2 = du0 * du2;
158  if (du0du1 > 0.0f && du0du2 > 0.0f) // same sign on all of them + not equal 0 ?
159  return -1; // no intersection occurs
160 
161  // compute plane of triangle1 and compute signed distances of triangle2 to the plane
162  const _Plane<value_t> plane1(triangle1.getVertexA(), triangle1.getVertexB(), triangle1.getVertexC());
163  float dv0 = plane1.planeTest(triangle2.getVertexA());
164  float dv1 = plane1.planeTest(triangle2.getVertexB());
165  float dv2 = plane1.planeTest(triangle2.getVertexC());
166 
167  // coplanarity robustness check
168  if (std::fabs(dv0) < epsilon)
169  dv0 = 0.0;
170  if (std::fabs(dv1) < epsilon)
171  dv1 = 0.0;
172  if (std::fabs(dv2) < epsilon)
173  dv2 = 0.0;
174 
175  float dv0dv1 = dv0 * dv1;
176  float dv0dv2 = dv0 * dv2;
177  if (dv0dv1 > 0.0f && dv0dv2 > 0.0f) // same sign on all of them + not equal 0 ?
178  return -1; // no intersection occurs
179 
180  // compute direction of intersection line
181  auto dir = plane1.getNormal().cross(plane2.getNormal());
182 
183  // compute and index to the largest component of D
184  value_t max = std::fabs(dir[0]);
185  uint8_t index = 0;
186  {
187  value_t b = std::fabs(dir[1]);
188  value_t c = std::fabs(dir[2]);
189  if (b > max)
190  max = b, index = 1;
191  if (c > max)
192  max = c, index = 2;
193  }
194 
195  // this is the simplified projection onto L
196  value_t pu0 = triangle1.getVertexA()[index];
197  value_t pu1 = triangle1.getVertexB()[index];
198  value_t pu2 = triangle1.getVertexC()[index];
199 
200  value_t pv0 = triangle2.getVertexA()[index];
201  value_t pv1 = triangle2.getVertexB()[index];
202  value_t pv2 = triangle2.getVertexC()[index];
203 
204  // compute interval for triangle 1
205  float intersectionA1;
206  _Vec3<value_t> pointA1;
207  float intersectionA2;
208  _Vec3<value_t> pointA2;
209  if (_computeIntersectionInterval(triangle1, pu0, pu1, pu2, du0, du1, du2, du0du1, du0du2, &intersectionA1,
210  &intersectionA2, pointA1, pointA2)) {
211  // first project onto an axis-aligned plane, that maximizes the area of the triangles
212  uint8_t i0, i1;
213  {
214  _Vec3<value_t> a = plane1.getNormal().getAbs();
215  if (a.x() > a.y()) {
216  if (a.x() > a.z()) {
217  i0 = 1; // x is greatest
218  i1 = 2;
219  } else {
220  i0 = 0; // z is greatest
221  i1 = 1;
222  }
223  } else {
224  if (a.z() > a.y()) {
225  i0 = 0; // z is greatest
226  i1 = 1;
227  } else {
228  i0 = 0; // y is greatest
229  i1 = 2;
230  }
231  }
232  }
233  _Vec2<value_t> U0(triangle1.getVertexA()[i0], triangle1.getVertexA()[i1]);
234  _Vec2<value_t> U1(triangle1.getVertexB()[i0], triangle1.getVertexB()[i1]);
235  _Vec2<value_t> U2(triangle1.getVertexC()[i0], triangle1.getVertexC()[i1]);
236  _Vec2<value_t> V0(triangle2.getVertexA()[i0], triangle2.getVertexA()[i1]);
237  _Vec2<value_t> V1(triangle2.getVertexB()[i0], triangle2.getVertexB()[i1]);
238  _Vec2<value_t> V2(triangle2.getVertexC()[i0], triangle2.getVertexC()[i1]);
239 
240  // test all edges of triangle 1 against the edges of triangle 2
241  EDGE_AGAINST_TRI_EDGES(V0, V1, U0, U1, U2);
242  EDGE_AGAINST_TRI_EDGES(V1, V2, U0, U1, U2);
243  EDGE_AGAINST_TRI_EDGES(V2, V0, U0, U1, U2);
244 
245  // finally, test if tri1 is totally contained in tri2 or vice versa
246  POINT_IN_TRI(V0, U0, U1, U2)
247  POINT_IN_TRI(U0, V0, V1, V2)
248  return -1;
249  }
250 
251  // compute interval for triangle 2
252  float intersectionB1 = 0.0f;
253  _Vec3<value_t> pointB1;
254  float intersectionB2 = 0.0f;
255  _Vec3<value_t> pointB2;
256  _computeIntersectionInterval(triangle2, pv0, pv1, pv2, dv0, dv1, dv2, dv0dv1, dv0dv2, &intersectionB1,
257  &intersectionB2, pointB1, pointB2);
258 
259  if (intersectionA1 > intersectionA2) {
260  std::swap(intersectionA1, intersectionA2);
261  std::swap(pointA1, pointA2);
262  }
263  if (intersectionB1 > intersectionB2) {
264  std::swap(intersectionB1, intersectionB2);
265  std::swap(pointB1, pointB2);
266  }
267 
268  if (intersectionA2 < intersectionB1 || intersectionB2 < intersectionA1)
269  return -1;
270 
271  // at this point, we know that the triangles intersect
272  if (intersectionB1 < intersectionA1)
273  tSegment.setFirstPoint(pointA1);
274  else
275  tSegment.setFirstPoint(pointB1);
276 
277  if (intersectionB2 < intersectionA2)
278  tSegment.setSecondPoint(pointB2);
279  else
280  tSegment.setSecondPoint(pointA2);
281  return true;
282 }
283 }
