46  if (a.size() != b.size())

48

50  const value_t * vecB = b.data();
51  const value_t * vecA_end = vecA + a.size();
52  while (vecA != vecA_end) {
53  if (*vecA != *vecB)
54  return *vecA < *vecB;

56  vecB++;
57  }
58  return false;

479  float_t length(uint32_t norm) const {
480  value_t ret = static_cast<value_t>(0);
481  switch (norm) {
482  case MAXIMUM_NORM: // 0
483  for (const auto & it : vec) {
484  ret = std::max(std::abs(it), ret);
485  }
486  return ret;
487  case MANHATTAN_NORM: // 1
488  for (const auto & it : vec) {
489  ret += std::abs(it);
490  }
491  return ret;
492  case EUCLIDEAN_NORM: // 2
493  for (const auto & it : vec) {
494  ret += it * it;
495  }
496  ret = std::sqrt(ret);
497  return ret;
498  default: // >= 3
499  if (norm % 2 == 0) { // exponent is even --> std::abs on base IS NOT necessary
500  for (const auto & it : vec) {
501  ret += std::pow(it, static_cast<float_t>(norm));
502  }
503  ret = std::pow(ret, static_cast<float_t>(1) / static_cast<float_t>(norm));
504  } else { // exponent is odd --> std::abs on base IS necessary (this works for all cases except maximum
505  // norm)
506  for (const auto & it : vec) {
507  ret += std::pow(std::abs(it), static_cast<float_t>(norm));
508  }
509  ret = std::pow(ret, static_cast<float_t>(1) / static_cast<float_t>(norm));
510  }
511  return ret;
512  }
