190  std::pair<bool, vec4_t> calcBarycentricCoordinates(const vec3_t & _p) const {
191  const vec4_t p(_p, 1);
192  const vec4_t v0(getVertexA(), 1);
193  const vec4_t v1(getVertexB(), 1);
194  const vec4_t v2(getVertexC(), 1);
195  const vec4_t v3(getVertexD(), 1);
196  const value_t det = matrix4x4_t(v0, v1, v2, v3).det();
197  if (det == 0) { // degenerate tetrahedron
198  return std::make_pair(false, vec4_t());
199  }
200  const value_t d0 = matrix4x4_t(p, v1, v2, v3).det();
201  const value_t d1 = matrix4x4_t(v0, p, v2, v3).det();
202  const value_t d2 = matrix4x4_t(v0, v1, p, v3).det();
203  const value_t d3 = matrix4x4_t(v0, v1, v2, p).det();
204 
205  return std::make_pair(true, vec4_t(d0 / det, d1 / det, d2 / det, d3 / det));
206  }

244  value_t distanceSquared(const vec3_t & point) const {
245  // Implementation taken from
246  // Christer Ericson: Real-Time Collision Detection. Morgan Kaufmann, 2004.
247 
248  // Start out assuming point inside all halfspaces, so closest to itself
249  value_t bestSquaredDist = std::numeric_limits<value_t>::max();
250 
251  // If point outside face abc then compute closest point on abc
252  if (pD.planeTest(point) >= 0.0f) {
253  const value_t squaredDist = getFaceD().distanceSquared(point);
254  // Update best closest point if (squared) distance is less than current best
255  if (squaredDist < bestSquaredDist) {
256  bestSquaredDist = squaredDist;
257  }
258  }
259  // Repeat test for face acd
260  if (pB.planeTest(point) >= 0.0f) {
261  const value_t squaredDist = getFaceB().distanceSquared(point);
262  if (squaredDist < bestSquaredDist) {
263  bestSquaredDist = squaredDist;
264  }
265  }
266  // Repeat test for face adb
267  if (pC.planeTest(point) >= 0.0f) {
268  const value_t squaredDist = getFaceC().distanceSquared(point);
269  if (squaredDist < bestSquaredDist) {
270  bestSquaredDist = squaredDist;
271  }
272  }
273  // Repeat test for face bdc
274  if (pA.planeTest(point) >= 0.0f) {
275  const value_t squaredDist = getFaceA().distanceSquared(point);
276  if (squaredDist < bestSquaredDist) {
277  bestSquaredDist = squaredDist;
278  }
279  }
280  if (bestSquaredDist == std::numeric_limits<value_t>::max()) {
281  bestSquaredDist = 0;
282  }
283  return bestSquaredDist;
284  }
