128  _Sphere & include(const vec3_t & position) {
129  // Simply include the point if this sphere is invalid
130  if (!isValid()) {
131  center = position;
132  radius = 0;
133  return *this;
134  }
135 
136  const value_t distCenterPosSquared = center.distanceSquared(position);
137  if (distCenterPosSquared <= radius * radius) {
138  return *this;
139  }
140  const value_t distCenterPos = std::sqrt(distCenterPosSquared);
141 
142  const value_t distSpherePos = distCenterPos - radius;
143  const value_t halfDistSpherePos = distSpherePos / 2;
144  // Move center halfDistSpherePos units into normalized (divide by length distCenterPos) direction.
145  center += (position - center) / distCenterPos * halfDistSpherePos;
146  radius += halfDistSpherePos;
147 
148  return *this;
149  }

152  _Sphere & include(const _Sphere & other) {
153  // Do not change the sphere if the other sphere is invalid
154  if (!other.isValid()) {
155  return *this;
156  }
157 
158  // Simply copy the other sphere if this sphere is invalid
159  if (!isValid()) {
160  *this = other;
161  return *this;
162  }
163 
164  const value_t centerDistance = center.distance(other.center);
165  if (centerDistance + other.radius <= radius) {
166  // This sphere already includes the other sphere
167  return *this;
168  }
169  if (centerDistance + radius <= other.radius) {
170  // The other sphere already includes this sphere
171  *this = other;
172  return *this;
173  }
174 
175  const value_t newRadius = (radius + centerDistance + other.radius) / 2;
176  /*
177  * Let n be the normalized vector between the two centers:
178  * n = (other.center - center) / centerDistance
179  * Calculate the outermost intersection t of the line defined by the two centers with this sphere.
180  * t = center - radius * n
181  * The new center can then be calculated as:
182  * center' = t + newRadius * n
183  * = center + (newRadius - radius) * n
184  * = center + ((newRadius - radius) / centerDistance) * (other.center - center)
185  */
186  const value_t proportion = (newRadius - radius) / centerDistance;
187  center += (other.center - center) * proportion;
188  radius = newRadius;
189 
190  return *this;
191  }
