171 template <typename Point_t>
172 inline bool PointOctree<Point_t>::insert(const Point_t & point) {
173  // make sure point is within boundary
174  if (!box.contains(point.getPosition())) {
175  return false;
176  } else if (isLeaf()) {
177  // add new point
178  points.push_back(point);
179 
180  // need to split?
181  if (points.size() > maxNumPoints && box.getExtentMax() >= minBoxSize * 2.0) {
182  const auto newBoxes = Helper::splitBoxCubeLike(box);
183  children.reserve(newBoxes.size());
184  for (const auto & newBox : newBoxes) {
185  children.emplace_back(newBox, minBoxSize, maxNumPoints);
186  }
187 
188  // distribute existing points to new children
189  for (const auto & currentPoint : points) {
190  for (auto & i : children) {
191  if (i.insert(currentPoint)) {
192  break;
193  }
194  }
195  }
196  points.clear();
197  }
198  return true;
199  } else {
200  PointOctree * leaf = findLeafCell(point.getPosition());
201  return leaf ? leaf->insert(point) : false;
202  }

250 template <typename Point_t>
251 inline void PointOctree<Point_t>::collectPointsWithinSphere(const Sphere_f & sphere, std::deque<Point_t> & out) const {
252  std::stack<const PointOctree *> activeCells;
253  activeCells.push(this);
254 
255  const float radius0_5 = 0.5f * sphere.getRadius();
256  while (!activeCells.empty()) {
257  const PointOctree * cell = activeCells.top();
258  activeCells.pop();
259  if (sphereBoxIntersection(sphere, cell->getBox())) {
260  if (cell->getBox().getExtentMax() < radius0_5) { // small box
261  bool inSphere = true;
262  for (uint_fast8_t i = 0; i < 8; ++i) { // completely inside sphere
263  if (sphere.isOutside(cell->getBox().getCorner(static_cast<corner_t>(i)))) {
264  inSphere = false;
265  break;
266  }
267  }
268  if (inSphere) {
269  cell->collectPoints(out);
270  continue;
271  }
272  }
273  if (cell->isLeaf()) {
274  for (const auto & p : cell->points) {
275  if (!sphere.isOutside(p.getPosition())) {
276  out.push_back(p);
277  }
278  }
279  } else {
280  // inner node
281  for (const auto & i : cell->children) {
282  activeCells.push(&i);
283  }
284  }
285  }
286  }

289 template <typename Point_t>
290 inline bool PointOctree<Point_t>::sphereBoxIntersection(const Sphere_f & sphere, const Box & box) {
291  float dist = 0.0f;
292  // find the square of the distance from the sphere to the box
293  if (sphere.getCenter().x() < box.getMinX()) {
294  const float s = sphere.getCenter().x() - box.getMinX();
295  dist += s * s;
296  } else if (sphere.getCenter().x() > box.getMaxX()) {
297  const float s = sphere.getCenter().x() - box.getMaxX();
298  dist += s * s;
299  }
300  if (sphere.getCenter().y() < box.getMinY()) {
301  const float s = sphere.getCenter().y() - box.getMinY();
302  dist += s * s;
303  } else if (sphere.getCenter().y() > box.getMaxY()) {
304  const float s = sphere.getCenter().y() - box.getMaxY();
305  dist += s * s;
306  }
307  if (sphere.getCenter().z() < box.getMinZ()) {
308  const float s = sphere.getCenter().z() - box.getMinZ();
309  dist += s * s;
310  } else if (sphere.getCenter().z() > box.getMaxZ()) {
311  const float s = sphere.getCenter().z() - box.getMaxZ();
312  dist += s * s;
313  }
314  return dist <= sphere.getRadius() * sphere.getRadius();
315 }

317 template <typename Point_t>
318 inline void PointOctree<Point_t>::getClosestPoints(const Vec3f & pos, size_t count, std::deque<Point_t> & out) const {
319  const PointOctree * leaf = findLeafCell(pos);
320  if (leaf == nullptr) {
321  return;
322  }
323  if (count == 0) {
324  out.clear();
325  return;
326  }
327  const float maxRadius = box.getDiameter();
328  for (float radius = leaf->box.getDiameter() * 0.25; radius <= maxRadius; radius *= 2.0) {
329  out.clear();
330  collectPointsWithinSphere(Sphere_f(pos, radius), out);
331  if (out.size() >= count)
332  break;
333  }
334  if (out.size() > count) {
335  std::vector<std::pair<float, size_t>> sortedPoints;
336 
337  // store (distance,pointIndex) in sortedPoints
338  size_t i = 0;
339  for (auto & point : out) {
340  sortedPoints.emplace_back(pos.distanceSquared(point.getPosition()), i++);
341  }
342  std::nth_element(sortedPoints.begin(), std::next(sortedPoints.begin(), (count - 1)), sortedPoints.end());
343  // keep only the 'count'-many closest points
344  sortedPoints.erase(std::next(sortedPoints.begin(), count), sortedPoints.end());
345 
346  // move the closest points to 'out'
347  std::deque<Point_t> result;
348  for (const auto & distancePointPair : sortedPoints) {
349  result.emplace_back(out[distancePointPair.second]);
350  }
351  using std::swap;
352  swap(out, result);
353  }
354 }

358  std::deque<Point_t> closestPoints;

362

364  size_t i = 0;

367
