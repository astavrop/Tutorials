129 template <typename _T>
130 inline _Vec3<_T> project(const _Vec3<_T> & obj, const _Matrix4x4<_T> & worldToScreen, const _Rect<_T> & viewport) {
131  const _T one = static_cast<_T>(1.0);
132  const _T two = static_cast<_T>(2.0);
133  const _Vec4<_T> v(obj, one);
134  _Vec4<_T> v2 = worldToScreen * v;
135  const _T w = v2.getW();
136  if (w == static_cast<_T>(0.0)) {
137  return _Vec3<_T>();
138  }
139  v2 /= w;
140  _Vec3<_T> win;
141  win.setX(viewport.getX() + (viewport.getWidth() * (v2.getX() + one)) / two);
142  win.setY(viewport.getY() + (viewport.getHeight() * (v2.getY() + one)) / two);
143  win.setZ((v2.getZ() + one) / two);
144  return win;
145 }

157 template <typename _T>
158 inline _Vec3<_T> unProject(const _Vec3<_T> & win, const _Matrix4x4<_T> & worldToScreen, const _Rect<_T> & viewport) {
159  const _T one = static_cast<_T>(1.0);
160  const _T two = static_cast<_T>(2.0);
161  _Vec4<_T> v;
162  v.setX((two * (win.getX() - viewport.getX())) / viewport.getWidth() - one);
163  v.setY((two * (win.getY() - viewport.getY())) / viewport.getHeight() - one);
164  v.setZ(two * win.getZ() - one);
165  v.setW(one);
166  _Vec4<_T> obj = worldToScreen.inverse() * v;
167  const _T w = obj.getW();
168  if (w == static_cast<_T>(0.0)) {
169  return _Vec3<_T>();
170  }
171  return _Vec3<_T>(obj.getX() / w, obj.getY() / w, obj.getZ() / w);
172 }
